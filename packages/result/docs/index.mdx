---
title: Quick Start
description: Explicit error handling with Ok/Err, safe helpers, and pipelining.
---


```bash
pnpm add @rectangular-labs/result
```

```ts
import { ok, err, Result } from "@rectangular-labs/result";

function double(n: number): Result<number, string> {
  return ok(n * 2);
}
```
